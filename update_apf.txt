The Audio Processing Framework (APF)

prerequisites: git subtree (included in git >= 1.7.11, but only in "contrib")

git remote add -f apf https://username@dev.qu.tu-berlin.de/git/apf-internal

# only first time:
git subtree add --prefix=apf --squash apf/master

git subtree pull --prefix=apf --squash apf/master

git subtree split --prefix=apf --annotate='(SSR) ' --branch ssr-changes
git push apf ssr-changes:master
